DEV (quick, use only for testing)

service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

Recommended (chat-safe) rules

service cloud.firestore {
  match /databases/{database}/documents {

    // users collection: user can read their own profile and write their own profile
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // rooms: doc contains participants array of uids
    match /rooms/{roomId} {
      // read: only participants
      allow read: if request.auth != null && request.auth.uid in resource.data.participants;
      // create: must include participants and creator must be in participants
      allow create: if request.auth != null && request.resource.data.participants is list && request.auth.uid in request.resource.data.participants;
      // update/delete: only participants
      allow update, delete: if request.auth != null && request.auth.uid in resource.data.participants;

      // messages subcollection: participants can read and create
      match /messages/{messageId} {
        allow read, create: if request.auth != null && request.auth.uid in get(/databases/$(database)/documents/rooms/$(roomId)).data.participants;
        // prevent editing/deleting messages
        allow update, delete: if false;
      }
    }

    // default: deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

--------------------------

How to apply these rules:
1) Open Firebase Console -> Firestore Database -> Rules
2) Replace existing rules with one of the blocks above
3) Click Publish

Notes:
- For quick debugging use the DEV rules, then switch to the Recommended rules before production.
- If you still see permission-denied errors after publishing, check that the user is actually authenticated (FirebaseAuth.currentUser != null) and inspect the app logs for detailed error messages.
